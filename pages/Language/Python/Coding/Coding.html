MySQL Cursor in Python

access/query results from a MySQL cursor inside a Python function

import mysql.connector

##Option 1: Using dictionary=True to get results as dictionaries
def get_users(conn):
    cursor = conn.cursor(dictionary=True)
    cursor.execute("SELECT id, name, email FROM users")
    for row in cursor.fetchall():
        print(row["id"], row["name"], row["email"])

##Option 2: Using column names to access results as tuples
def get_users(conn):
    cursor = conn.cursor()
    cursor.execute("SELECT id, name, email FROM users")
    columns = [desc[0] for desc in cursor.description]
    for row in cursor.fetchall():
        row_dict = dict(zip(columns, row))
        print(row_dict["id"], row_dict["name"], row_dict["email"])

##Option 3: Using namedtuple for more structured access
from collections import namedtuple
def get_users(conn):
    cursor = conn.cursor()
    cursor.execute("SELECT id, name, email FROM users") 
    User = namedtuple("User", [desc[0] for desc in cursor.description])
    for row in cursor.fetchall():
        user = User(*row)
        print(user.id, user.name, user.email)

Flask + MySQL

Here’s how to properly connect your cursor → Python → JSON response:

from flask import jsonify

###Example Flask route to get users from MySQL and return as JSON
def get_users(conn):
    cursor = conn.cursor(dictionary=True)
    cursor.execute("SELECT id, name FROM users")
    rows = cursor.fetchall()
    return jsonify(rows)

##Example Flask route to get a single user by ID and return as JSON
def get_user(conn, user_id):
    cursor = conn.cursor(dictionary=True)
    cursor.execute("SELECT id, name FROM users WHERE id = %s", (user_id,))
    row = cursor.fetchone()
    return jsonify(row)

##Example Flask route to get all users and return a structured JSON response
def get_users(conn):
    cursor = conn.cursor(dictionary=True)
    cursor.execute("SELECT id, name FROM users")
    return jsonify({
        "status": "success",
        "data": cursor.fetchall()
    })
